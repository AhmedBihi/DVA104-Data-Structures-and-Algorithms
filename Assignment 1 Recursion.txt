#include <stdio.h>
#include <stdlib.h>

int recursion(int n)
{
    int f;
    static int i = 0;
    i++;
    printf("%d: sum(%d)\n", i, n);
    if(n==1)
    {
        printf("Basfall!\n");
        printf("%d: return %d\n",i, 1);
        return 1;
    }
    else
    {
        f = n + rekursion(n-1);
    }

    i--;
    printf("%d: return %d\n",i, f);
    return f;
}

int main()
{
    int sum, n;

    printf("Return sum 1 + 2 + 3... + n\n");
    printf("n: ");
    scanf("%d", &n);

    sum = recursion(n);

    printf("Sum: %d", sum);
    return 0;
}
